<script>

  var tg = window.Telegram.WebApp;

  tg.ready();

  tg.expand();


  var user = tg.initDataUnsafe && tg.initDataUnsafe.user;


  var userEl = document.getElementById('user-info');

  if (user) {

    var name = [user.first_name, user.last_name].filter(Boolean).join(' ') || 'Пользователь';

    userEl.innerHTML = '👤 <strong>' + name + '</strong> &nbsp;·&nbsp; ID: ' + user.id;

  } else {

    userEl.textContent = '👤 Пользователь';

  }


  // Тройной тап — показать отладку

  var taps = 0;

  userEl.addEventListener('click', function() {

    taps++;

    if (taps >= 3) {

      taps = 0;

      var d = document.getElementById('debug');

      d.style.display = 'block';

      d.textContent =

        'platform: ' + tg.platform + '\n' +

        'version: ' + tg.version + '\n' +

        'initData пустой: ' + (!tg.initData) + '\n' +

        'user: ' + JSON.stringify(user);

    }

  });


  var labels = {

    plomb: '🔍 Запускаю обработку пломб...',

    guarantee: '📋 Запускаю реестр гарантии...',

    scan: '📦 Запускаю сканирование...',

    template: '📎 Запускаю загрузку шаблона...',

    qr: '📲 Запускаю генерацию QR...',

    supply: '🚚 Запускаю новую поставку...',

    help: '❓ Открываю справку...'

  };


  function sendAction(action) {

    if (tg.HapticFeedback) tg.HapticFeedback.impactOccurred('light');


    document.getElementById('overlay-text').textContent = labels[action] || 'Отправляю...';

    document.getElementById('overlay').classList.add('active');


    setTimeout(function() {

      try {

        var data = JSON.stringify({

          action: action,

          user_id: user ? user.id : null,

          timestamp: Date.now()

        });

        tg.sendData(data);

        // sendData закрывает Mini App автоматически.

        // Если через 4 сек ничего не случилось — показываем ошибку.

        setTimeout(function() {

          document.getElementById('overlay').classList.remove('active');

          var t = document.getElementById('toast');

          t.textContent = '❌ sendData не сработал. Нажмите 3 раза на имя внизу для диагностики.';

          t.classList.add('active');

          setTimeout(function() { t.classList.remove('active'); }, 6000);

        }, 4000);

      } catch(e) {

        document.getElementById('overlay').classList.remove('active');

        var t = document.getElementById('toast');

        t.textContent = '❌ Ошибка: ' + e.message;

        t.classList.add('active');

        setTimeout(function() { t.classList.remove('active'); }, 6000);

      }

    }, 250);

  }

</script>
