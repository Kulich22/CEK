    <script>

        // Инициализация Telegram Web App

        const tg = window.Telegram.WebApp;

        

        // Сообщаем Telegram, что приложение готово

        tg.ready();

        

        // Растягиваем на всю высоту

        tg.expand();

        

        // Получаем информацию о пользователе

        const user = tg.initDataUnsafe?.user;

        if (user) {

            document.getElementById('user-info').innerHTML = 

                `👤 ${user.first_name} ${user.last_name || ''}<br>ID: ${user.id}`;

        } else {

            document.getElementById('user-info').innerHTML = '👤 Пользователь не определён';

        }

        

        // Функция отправки команды в бота

        function sendCommand(command) {

            // Создаем кнопку "Отправить" в Telegram

            tg.MainButton.setText(`Выбрано: ${command}`);

            tg.MainButton.show();

            

            // При нажатии на кнопку отправляем данные в бота

            tg.MainButton.onClick(function() {

                tg.sendData(JSON.stringify({

                    action: command,

                    user_id: user?.id,

                    timestamp: Date.now()

                }));

                tg.close();

            });

            

            // Показываем уведомление

            tg.HapticFeedback.impactOccurred('medium');

        }

        

        // Обработка кнопки "Назад" в Telegram

        tg.onEvent('backButtonClicked', function() {

            tg.close();

        });
